async function sleep(ms) {

async function run() {
  const browser = await chromium.launch({ headless: true });
  const page = await browser.newPage();
  await page.setViewportSize({ width: 1920, height: 1080 });

  const errors = [];
  const consoleErrors = [];
  page.on('console', msg => {
    if (msg.type() === 'error') consoleErrors.push(msg.text());
  });

  await page.goto(HTML_PATH);
  // Wait for Reveal.js to initialize (Reveal dispatches 'ready' event)
  await page.waitForFunction(() => window.Reveal && window.Reveal.isReady(), { timeout: 10000 });

  // ── CHECK 1: Reveal.js configuration ──
  const config = await page.evaluate(() => ({
    transition: Reveal.getConfig().transition,
    transitionSpeed: Reveal.getConfig().transitionSpeed,
    progress: Reveal.getConfig().progress,
    keyboard: Reveal.getConfig().keyboard,
    mouseWheel: Reveal.getConfig().mouseWheel,
    fragments: Reveal.getConfig().fragments,
    autoSlide: Reveal.getConfig().autoSlide,
    autoSlideStoppable: Reveal.getConfig().autoSlideStoppable,
  }));

  if (config.transition !== 'fade')
    errors.push(`Transition: expected 'fade', got '${config.transition}'`);
  if (config.transitionSpeed !== 'slow')
    errors.push(`TransitionSpeed: expected 'slow', got '${config.transitionSpeed}'`);
  if (config.progress !== false)
    errors.push(`Progress: expected false, got ${config.progress}`);
  if (config.keyboard !== true)
    errors.push(`Keyboard: expected true`);
  if (config.mouseWheel !== true)
    errors.push(`MouseWheel: expected true`);
  if (config.autoSlide !== 0)
    errors.push(`AutoSlide: expected 0 (disabled), got ${config.autoSlide}`);

  // ── CHECK 2: Slide count ──
  const slideCount = await page.evaluate(() =>
    document.querySelectorAll('.reveal .slides > section').length
  );
  if (slideCount < 31) errors.push(`Slide count: expected ≥ 31, got ${slideCount}`);

  // ── CHECK 3: No vertical (nested) slides ──
  const nestedCount = await page.evaluate(() =>
    document.querySelectorAll('.reveal .slides > section > section').length
  );
  if (nestedCount > 0) errors.push(`Found ${nestedCount} nested sections — all must be horizontal`);

  // ── CHECK 4: Fullscreen button exists ──
  const fsBtn = await page.$('#fullscreen-btn');
  if (!fsBtn) errors.push('Fullscreen button #fullscreen-btn not found');

  // ── CHECK 5: CSS variables ──
  const cssVars = await page.evaluate(() => {
    const style = getComputedStyle(document.documentElement);
    return {
      bgPrimary: style.getPropertyValue('--bg-primary').trim(),
      bgSecondary: style.getPropertyValue('--bg-secondary').trim(),
      textPrimary: style.getPropertyValue('--text-primary').trim(),
      textSecondary: style.getPropertyValue('--text-secondary').trim(),
      accent: style.getPropertyValue('--accent').trim(),
      success: style.getPropertyValue('--success').trim(),
      danger: style.getPropertyValue('--danger').trim(),
      border: style.getPropertyValue('--border').trim(),
    };
  });
  if (cssVars.bgPrimary !== '#0a0f1a') errors.push(`--bg-primary: expected #0a0f1a, got ${cssVars.bgPrimary}`);
  if (cssVars.accent !== '#ff6b35') errors.push(`--accent: expected #ff6b35, got ${cssVars.accent}`);
  if (cssVars.success !== '#22c55e') errors.push(`--success: expected #22c55e, got ${cssVars.success}`);
  if (cssVars.danger !== '#ef4444') errors.push(`--danger: expected #ef4444, got ${cssVars.danger}`);
  if (cssVars.bgSecondary !== '#111827') errors.push(`--bg-secondary: expected #111827, got ${cssVars.bgSecondary}`);
  if (cssVars.textPrimary !== '#f9fafb') errors.push(`--text-primary: expected #f9fafb, got ${cssVars.textPrimary}`);
  if (cssVars.textSecondary !== '#9ca3af') errors.push(`--text-secondary: expected #9ca3af, got ${cssVars.textSecondary}`);
  if (cssVars.border !== '#1f2937') errors.push(`--border: expected #1f2937, got ${cssVars.border}`);

  // ── CHECK 6: Progress bar hidden ──
  const progressVisible = await page.evaluate(() => {
    const el = document.querySelector('.reveal .progress');
    return el && window.getComputedStyle(el).display !== 'none';
  });
  if (progressVisible) errors.push('Progress bar should be hidden');

  // ── CHECK 7: Navigate all slides, verify per-slide content, no TBD/TODO ──
  const SLIDE_CONTENT = [
    // Slide 0: 主标题
    ['AI压缩了执行力', '执行力', '判断力'],
    // Slide 1: 副标题+来源
    ['值钱', '焦虑', 'Anthropic'],
    // Slide 2: 第一章标题
    ['第一章', '执行层的差距', '正在消失'],
    // Slide 3: Ch1 hook
    ['从没写过Python', '会计'],
    // Slide 4: 89% vs 88%
    ['89%', '88%', '软件工程师', '其他职业'],
    // Slide 5: staircase 15-33%
    ['15%', '28%', '33%', '新手', '中级', '高级'],
    // Slide 6: 放弃率
    ['19%', '5-7%', '放弃率', '碰壁'],
    // Slide 7: 翻盘 flow
    ['4%', '15%', '翻盘'],
    // Slide 8: 第二章标题
    ['第二章', '同一句话', '不同价格'],
    // Slide 9: Ch2 hook
    ['努力程度', '不一样'],
    // Slide 10: 动作数
    ['5', '12', '动作/指令'],
    // Slide 11: 输出词数
    ['600', '3,200', '词/指令', '5.3'],
    // Slide 12: EN/CN quote
    ['People decide what to build', '人决定造什么'],
    // Slide 13: 指令对比
    ['你帮我写个对账脚本', '处理三种异常情况'],
    // Slide 14: 第三章标题
    ['第三章', '梯子正在', '被抽掉'],
    // Slide 15: Ch3 hook
    ['传统白领', '成长路径'],
    // Slide 16: Dallas Fed columns
    ['可编码', '默会', '替代'],
    // Slide 17: Dallas Fed quote
    ['cost-ineffective', 'employee development'],
    // Slide 18: 修复代码 trend
    ['33% → 19%', '修复代码'],

    ['~10% → 20%', '写新代码', '数据分析'],

    ['第四章', '焦虑不是来自未知', '来自已知'],
    // Slide 21: Ch4 hook
    ['一边疯狂用AI', '一边疯狂焦虑'],
    // Slide 22: 93% vs 58%
    ['93%', '58%', '中国', '全球'],
    // Slide 23: 焦虑率
    ['40%', '59%', '年轻人', '替代'],
    // Slide 24: 本质洞察 + case
    ['越了解AI', '越焦虑', '广告公司策略总监', '裁了1/3'],
    // Slide 25: 第五章标题
    ['第五章', '判断力会不会', '也被压缩'],
    // Slide 26: 判断力分层
    ['执行判断', '价值判断', '性能够不够', '值不值得做'],
    // Slide 27: Anthropic quote
    ['not substituting', 'domain expertise'],
    // Slide 28: 金句
    ['写一千个方案', '就用这个'],
    // Slide 29: 结尾
    ['不敢交给它'],
    // Slide 30: 来源
    ['数据来源', 'Dallas Fed', 'KPMG', '澎湃新闻', 'Soul', '虎嗅'],
  ];

  for (let i = 0; i < slideCount; i++) {
    await page.evaluate((idx) => Reveal.slide(idx), i);
    await sleep(1000); // Wait for fade transition

    const slideText = await page.evaluate(() => document.body.innerText);

    // Check for no TBD/TODO
    if (/TBD|TODO|PLACEHOLDER/i.test(slideText)) {
      errors.push(`Slide ${i}: contains TBD/TODO placeholder`);
    }

    // Per-slide content verification
    if (SLIDE_CONTENT[i]) {
      for (const str of SLIDE_CONTENT[i]) {
        if (!slideText.includes(str)) {
          errors.push(`Slide ${i}: missing required string "${str}"`);
        }
      }
    }
  }

  // ── CHECK 8: All required strings present in full DOM ──
  const fullText = await page.evaluate(() => document.body.innerText);
  for (const str of REQUIRED_STRINGS) {
    if (!fullText.includes(str)) {
      errors.push(`Missing string: "${str}" not found in DOM`);
    }
  }

  // ── CHECK 9: Chapter titles ──
  for (const ch of CHAPTER_TITLES) {
    if (!fullText.includes(ch)) {
      errors.push(`Missing chapter title: "${ch}"`);
    }
  }

  // ── CHECK 10: No console errors ──
  if (consoleErrors.length > 0) {
    errors.push(`Console errors: ${JSON.stringify(consoleErrors)}`);
  }

  // ── CHECK 11: No overflow (scrollHeight > clientHeight) on any slide ──
  const overflowSlides = await page.evaluate(() => {
    const sections = document.querySelectorAll('.reveal .slides > section');
    const overflow = [];
    sections.forEach((s, i) => {
      if (s.scrollHeight > s.clientHeight + 2) {
        overflow.push(`Slide ${i}: overflow (scrollHeight=${s.scrollHeight}, clientHeight=${s.clientHeight})`);
      }
    });
    return overflow;
  });
  errors.push(...overflowSlides);

  // ── REPORT ──
  console.log(`\n=== TEST RESULTS ===`);
  console.log(`Slides: ${slideCount}`);
  console.log(`Errors: ${errors.length}`);
  if (errors.length > 0) {
    console.log('\nFAILURES:');
    errors.forEach(e => console.log(`  ❌ ${e}`));
    process.exit(1);
  } else {
    console.log('  ✅ All checks passed');
    process.exit(0);
  }
}

run().catch(err => {
  console.error('Test crashed:', err);
  process.exit(1);
});
